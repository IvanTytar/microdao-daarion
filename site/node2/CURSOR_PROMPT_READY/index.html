# Tool –¥–ª—è –ª–æ–∫–∞–ª—å–Ω–æ—ó –ø–∞–º'—è—Ç—ñ

    import httpx
    response = httpx.post(
        "http://localhost:9401/query",
        json={"query": query, "query_type": "vector_search", "limit": 10},
        timeout=30.0

    return response.json().get("results", [])

    allow_delegation=False
